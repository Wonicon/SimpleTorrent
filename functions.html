这里列出了所有文档化的结构体和联合体的成员变量，并附带结构或联合所属的文件:

- a -

  • addr : Peer , WaitPeer

- b -

  • begin : PeerMsg
  • bitfield : Peer , PeerMsg
  • bitfield_size : MetaInfo
  • block : PeerMsg
  • buf : HttpRequest

- c -

  • cancel : PeerMsg
  • contribution : Peer
  • curr : HttpRequest , State

- d -

  • d_key : BNode
  • d_next : BNode
  • d_val : BNode
  • delim : HttpRequest
  • direction : WaitPeer

- e -

  • end : BNode

- f -

  • fd : Peer , WaitPeer
  • file : MetaInfo
  • file_size : MetaInfo

- g -

  • get_choked : Peer
  • get_interested : Peer

- h -

  • hash : PieceInfo
  • have : PeerMsg
  • host : Tracker
  • hs_info_hash : PeerHandShake
  • hs_peer_id : PeerHandShake
  • hs_pstr : PeerHandShake
  • hs_pstrlen : PeerHandShake
  • hs_reserved : PeerHandShake

- i -

  • i : BNode
  • id : PeerMsg
  • index : PeerMsg
  • info_hash : MetaInfo
  • ip : Peer , WaitPeer
  • is_choked : Peer
  • is_downloaded : PieceInfo
  • is_interested : Peer

- l -

  • l_item : BNode
  • l_next : BNode
  • len : PeerMsg
  • length : PeerMsg
  • listen_fd : MetaInfo

- m -

  • method : Tracker
  • msg : Peer

- n -

  • nr_owners : PieceInfo
  • nr_peers : MetaInfo
  • nr_pieces : MetaInfo
  • nr_trackers : MetaInfo
  • nr_wait_peers : MetaInfo

- p -

  • peers : MetaInfo
  • piece : PeerMsg
  • piece_index : PeerMsg
  • piece_size : MetaInfo
  • pieces : MetaInfo
  • port : MetaInfo , Peer , Tracker , WaitPeer

- r -

  • request : PeerMsg , Tracker
  • requested_pieces : Peer
  • requested_subpieces : Peer
  • requesting_begin : Peer
  • requesting_index : Peer

- s -

  • s_data : BNode
  • s_size : BNode
  • sfd : Tracker
  • speed : Peer
  • start : BNode , State
  • start_time : Peer
  • sub_count : MetaInfo
  • sub_size : MetaInfo
  • substate : PieceInfo
  • subtimer : PieceInfo

- t -

  • timerfd : MetaInfo , Tracker
  • trackers : MetaInfo
  • type : BNode

- w -

  • wait_peers : MetaInfo
  • wanted : Peer
